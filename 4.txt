int f4(int n) { // ตรวจสอบค่า n มีมากกว่า1 หรือไม่
    if (n > 1) // ถ้า n มากกว่า 1
        return 1 + 2 * f4(n- 1); // โปรเเกรมจะทำการเรียกฟังก์ชั่น f4 อีกครั้งโดยลดค่า n ลง1 จาก n - 1
    return 0; // n ลดไปเรื่อยๆ จนถึง 1 จะคืนค่าเป็น 0
}
// O(n)
